SELECT spcdc.id as record_number, bld.name as building_name, flr.name as floor_name, flt.name as flat_name, rom.name as room_name,dvc.name as lpt_comm_device_name, eqpmt.name as equipment_name, eqpmt1.name as equipment_name1
	  
             FROM smsserver_parallel_comm_device_configuration spcdc, building bld, floor flr, flat flt,room rom, device dvc, equipment eqpmt, equipment eqpmt1 WHERE spcdc.building_id = bld.id and spcdc.floor_id = flr.id and spcdc.flat_id = flt.id and spcdc.room_id = rom.id and spcdc.lpt_comm_device_id = dvc.id and spcdc.equipment1_id = eqpmt.id and spcdc.equipment2_id = eqpmt1.id
